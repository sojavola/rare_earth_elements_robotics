
import rclpy
from rclpy.node import Node
import numpy as np
import json
import time
from threading import Lock

from std_msgs.msg import String, Float32MultiArray
from geometry_msgs.msg import Pose2D
from visualization_msgs.msg import Marker, MarkerArray
from std_msgs.msg import ColorRGBA

class MineralDetectorNode(Node):
    def __init__(self):
        super().__init__('mineral_detector_node')
        
        self.robot_positions = {}
        self.mineral_map = None
        self.lock = Lock()
        
        # Détections actives
        self.active_detections = []
        self.detection_id = 0
        
        # Types de terres rares avec propriétés uniques
        self.ree_types = {
            0: {'name': 'REE_Oxides', 'color': (0.8, 0.2, 0.2), 'symbol': '🟥', 'rarity': 1.0},
            1: {'name': 'REE_Silicates', 'color': (0.2, 0.8, 0.2), 'symbol': '🟩', 'rarity': 1.2},
            2: {'name': 'REE_Phosphates', 'color': (0.2, 0.2, 0.8), 'symbol': '🟦', 'rarity': 1.5},
            3: {'name': 'REE_Carbonates', 'color': (0.8, 0.8, 0.2), 'symbol': '🟨', 'rarity': 1.8}
        }
        
        # Subscribers
        for i in range(4):
            self.create_subscription(
                Pose2D, f'/robot_{i}/position',
                self.create_robot_callback(i), 10
            )
        
        self.create_subscription(
            Float32MultiArray, '/mineral_map',
            self.mineral_callback, 10
        )
        
        # Publishers
        self.detection_pub = self.create_publisher(String, '/mineral_detections', 10)
        self.markers_pub = self.create_publisher(MarkerArray, '/mineral_detection_markers', 10)
        self.analysis_pub = self.create_publisher(String, '/mineral_analysis', 10)
        
        # Timer pour le traitement
        self.create_timer(0.3, self.process_detections)  # 3.3 Hz
        
        self.get_logger().info('🔍 Mineral Detector Node started')
    
    def create_robot_callback(self, robot_id):
        def callback(msg):
            with self.lock:
                self.robot_positions[robot_id] = (msg.x, msg.y, msg.theta)
        return callback
    
    def mineral_callback(self, msg):
        with self.lock:
            if len(msg.data) == 100 * 100 * 4:
                self.mineral_map = np.array(msg.data).reshape(100, 100, 4)
    
    def process_detections(self):
        if self.mineral_map is None:
            return
        
        with self.lock:
            current_time = time.time()
            
            # Nettoyer les anciennes détections
            self.active_detections = [
                d for d in self.active_detections 
                if current_time - d['timestamp'] < 8.0  # 8 secondes de durée de vie
            ]
            
            # Vérifier les nouvelles détections pour chaque robot
            for robot_id, position in self.robot_positions.items():
                if position is None:
                    continue
                
                x, y, theta = position
                grid_x, grid_y = int(x), int(y)
                
                if 0 <= grid_x < 100 and 0 <= grid_y < 100:
                    # Analyser la zone autour du robot (rayon de 3 cases)
                    detections = self.scan_area(grid_x, grid_y, robot_id, 3)
                    
                    for detection in detections:
                        if self.is_new_detection(detection):
                            self.active_detections.append(detection)
                            self.publish_detection(detection)
            
            # Publier les marqueurs de détection
            self.publish_detection_markers()
            
            # Publier l'analyse globale
            self.publish_analysis_report()
    
    def scan_area(self, center_x, center_y, robot_id, radius):
        """Scanne une zone carrée autour de la position du robot"""
        detections = []
        
        for y in range(center_y - radius, center_y + radius + 1):
            for x in range(center_x - radius, center_x + radius + 1):
                if 0 <= x < 100 and 0 <= y < 100:
                    concentrations = self.mineral_map[y, x, :]
                    max_conc = np.max(concentrations)
                    mineral_type = np.argmax(concentrations)
                    
                    if max_conc > 0.4:  # Seuil de détection
                        distance = np.sqrt((x - center_x)**2 + (y - center_y)**2)
                        
                        detection = {
                            'id': f"{robot_id}_{x}_{y}_{mineral_type}",
                            'robot_id': robot_id,
                            'position': (x, y),
                            'mineral_type': mineral_type,
                            'concentration': max_conc,
                            'distance': distance,
                            'signal_strength': max_conc / (distance + 1),
                            'timestamp': time.time(),
                            'value_score': max_conc * self.ree_types[mineral_type]['rarity']
                        }
                        detections.append(detection)
        
        return detections
    
    def is_new_detection(self, detection):
        """Vérifie si c'est une nouvelle détection"""
        for existing in self.active_detections:
            if (existing['position'] == detection['position'] and 
                existing['mineral_type'] == detection['mineral_type']):
                return False
        return True
    
    def publish_detection(self, detection):
        """Publie une détection individuelle"""
        ree_info = self.ree_types[detection['mineral_type']]
        
        detection_data = {
            'timestamp': self.get_clock().now().nanoseconds,
            'detection': {
                'robot_id': detection['robot_id'],
                'position': {'x': detection['position'][0], 'y': detection['position'][1]},
                'mineral_type': ree_info['name'],
                'symbol': ree_info['symbol'],
                'concentration': float(detection['concentration']),
                'signal_strength': float(detection['signal_strength']),
                'value_score': float(detection['value_score']),
                'rarity': ree_info['rarity'],
                'status': 'HIGH_SIGNAL' if detection['signal_strength'] > 0.3 else 'MEDIUM_SIGNAL' if detection['signal_strength'] > 0.15 else 'LOW_SIGNAL'
            }
        }
        
        msg = String()
        msg.data = json.dumps(detection_data)
        self.detection_pub.publish(msg)
        
        # Log coloré
        status_emoji = '🔴' if detection['signal_strength'] > 0.3 else '🟡' if detection['signal_strength'] > 0.15 else '🔵'
        self.get_logger().info(
            f'{status_emoji} {ree_info["symbol"]} Robot {detection["robot_id"]} '
            f'détecte {ree_info["name"]} - Force: {detection["signal_strength"]:.2f} '
            f'à ({detection["position"][0]}, {detection["position"][1]})'
        )
    
    def publish_detection_markers(self):
        """Publie les marqueurs de détection visuels"""
        marker_array = MarkerArray()
        
        for i, detection in enumerate(self.active_detections):
            ree_info = self.ree_types[detection['mineral_type']]
            
            # Marqueur principal
            marker = self.create_mineral_marker(detection, ree_info, i)
            marker_array.markers.append(marker)
            
            # Effet de particules (pour les hautes concentrations)
            if detection['concentration'] > 0.7:
                particle_marker = self.create_particle_effect(detection, ree_info, i + 1000)
                marker_array.markers.append(particle_marker)
            
            # Texte flottant (pour les très hautes valeurs)
            if detection['value_score'] > 1.0:
                text_marker = self.create_text_marker(detection, ree_info, i + 2000)
                marker_array.markers.append(text_marker)
        
        if marker_array.markers:
            self.markers_pub.publish(marker_array)
    
    def create_mineral_marker(self, detection, ree_info, marker_id):
        """Crée un marqueur visuel avancé pour un dépôt minéral"""
        marker = Marker()
        marker.header.frame_id = "map"
        marker.header.stamp = self.get_clock().now().to_msg()
        marker.ns = "mineral_deposits"
        marker.id = marker_id
        marker.type = Marker.SPHERE if detection['concentration'] > 0.6 else Marker.CUBE
        marker.action = Marker.ADD
        
        x, y = detection['position']
        marker.pose.position.x = float(x)
        marker.pose.position.y = float(y)
        
        # Hauteur basée sur la valeur
        base_height = 0.3 + (detection['value_score'] * 0.5)
        marker.pose.position.z = base_height
        marker.pose.orientation.w = 1.0
        
        # Taille basée sur la concentration
        base_size = 0.8 + (detection['concentration'] * 1.5)
        marker.scale.x = base_size
        marker.scale.y = base_size
        marker.scale.z = base_size * 0.7
        
        # Couleur avec effet de brillance
        r, g, b = ree_info['color']
        intensity = min(1.0, detection['concentration'] * 1.3)
        
        marker.color.r = r * intensity
        marker.color.g = g * intensity
        marker.color.b = b * intensity
        marker.color.a = 0.85
        
        # Animation de pulsation pour les hautes valeurs
        if detection['value_score'] > 0.8:
            pulse = (np.sin(time.time() * 3) + 1) * 0.1
            marker.scale.x += pulse
            marker.scale.y += pulse
        
        marker.lifetime.nanosec = int(3e9)  # 3 secondes
        
        return marker
    
    def create_particle_effect(self, detection, ree_info, marker_id):
        """Crée un effet de particules pour les dépôts importants"""
        marker = Marker()
        marker.header.frame_id = "map"
        marker.header.stamp = self.get_clock().now().to_msg()
        marker.ns = "particle_effects"
        marker.id = marker_id
        marker.type = Marker.POINTS
        marker.action = Marker.ADD
        
        x, y = detection['position']
        marker.pose.position.x = float(x)
        marker.pose.position.y = float(y)
        marker.pose.position.z = 0.5
        marker.pose.orientation.w = 1.0
        
        marker.scale.x = 0.1
        marker.scale.y = 0.1
        marker.scale.z = 0.0
        
        # Générer des points de particules
        points = []
        colors = []
        
        for i in range(8):  # 8 particules
            angle = (time.time() * 2 + i * 0.785) % (2 * np.pi)
            radius = (time.time() * 0.5) % 1.5
            
            px = radius * np.cos(angle)
            py = radius * np.sin(angle)
            pz = radius * 0.3
            
            points.append(Point(x=px, y=py, z=pz))
            
            # Couleur dégradée
            color = ColorRGBA()
            r, g, b = ree_info['color']
            color.r = r
            color.g = g
            color.b = b
            color.a = 0.6 - (radius / 1.5) * 0.4
            colors.append(color)
        
        marker.points = points
        marker.colors = colors
        marker.lifetime.nanosec = int(2e9)
        
        return marker
    
    def create_text_marker(self, detection, ree_info, marker_id):
        """Crée un marqueur de texte flottant"""
        marker = Marker()
        marker.header.frame_id = "map"
        marker.header.stamp = self.get_clock().now().to_msg()
        marker.ns = "value_text"
        marker.id = marker_id
        marker.type = Marker.TEXT_VIEW_FACING
        marker.action = Marker.ADD
        
        x, y = detection['position']
        marker.pose.position.x = float(x)
        marker.pose.position.y = float(y)
        marker.pose.position.z = 2.0 + (detection['value_score'] * 0.3)
        marker.pose.orientation.w = 1.0
        
        marker.scale.z = 0.8  # Taille du texte
        
        marker.color.r = 1.0
        marker.color.g = 1.0
        marker.color.b = 1.0
        marker.color.a = 0.9
        
        marker.text = f"{ree_info['symbol']} {detection['value_score']:.1f}"
        marker.lifetime.nanosec = int(3e9)
        
        return marker
    
    def publish_analysis_report(self):
        """Publie un rapport d'analyse global"""
        if not self.active_detections:
            return
        
        # Calculer les statistiques
        total_detections = len(self.active_detections)
        avg_value = np.mean([d['value_score'] for d in self.active_detections])
        max_value = np.max([d['value_score'] for d in self.active_detections]) if self.active_detections else 0
        
        # Compter par type
        type_counts = {}
        for detection in self.active_detections:
            type_name = self.ree_types[detection['mineral_type']]['name']
            type_counts[type_name] = type_counts.get(type_name, 0) + 1
        
        analysis_data = {
            'timestamp': self.get_clock().now().nanoseconds,
            'summary': {
                'total_active_detections': total_detections,
                'average_value_score': float(avg_value),
                'max_value_score': float(max_value),
                'detection_heat_level': 'HIGH' if total_detections > 10 else 'MEDIUM' if total_detections > 5 else 'LOW'
            },
            'type_breakdown': type_counts
        }
        
        msg = String()
        msg.data = json.dumps(analysis_data)
        self.analysis_pub.publish(msg)

def main():
    rclpy.init()
    node = MineralDetectorNode()
    try:
        rclpy.spin(node)
    except KeyboardInterrupt:
        node.get_logger().info('🛑 Shutting down Mineral Detector Node')
    finally:
        node.destroy_node()
        rclpy.shutdown()

if __name__ == '__main__':
    main()
